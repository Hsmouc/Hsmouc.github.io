- 3.6
- use a range for to change all the characters in a string to X.
{% highlight c %}
#include<iostream>
#include<string>
using namespace std;
int main()
{
    string str;
    cin<<str;
    for(auto &c : str) c = 'X';
    cout<<str<<endl;
    return 0;
}

{
    cout << "Enter a string of characters including punctuation." << endl;
